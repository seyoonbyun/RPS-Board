    <script>
        // 전역 변수
        let currentUser = null;

        // 페이지 로드 시 초기화

            // 로그인 폼

            // 로그아웃 버튼
            document.getElementById('logoutButton').addEventListener('click', handleLogout);
            
            // 데이터 폼

        // 로그인 처리
        function handleLogin(e) {
            e.preventDefault();
            
            const email = document.getElementById('loginEmail').value;
            const password = document.getElementById('loginPassword').value;
            
            if (password.length !== 4) {
                alert('비밀번호는 4자리여야 합니다.');
                return;
            }
            
            currentUser = { email, password };
            localStorage.setItem('bni_current_user', JSON.stringify(currentUser));
            
            loadUserData();
            showMainPage();
            showToast('로그인되었습니다!');
        }

        // 로그아웃 처리
        function handleLogout() {
            currentUser = null;
            localStorage.removeItem('bni_current_user');
            showLoginPage();
            showToast('로그아웃되었습니다!');
        }

        // 사용자 데이터 로드
        function loadUserData() {
            const savedData = localStorage.getItem(`bni_data_${currentUser.email}`);
            if (savedData) {
                userData = JSON.parse(savedData);
                populateForm();
            } else {
                userData = {};
            }
        }

        // 폼에 데이터 채우기
        function populateForm() {
            Object.keys(userData).forEach(key => {
                const element = document.getElementById(key);
                if (element) {
                    element.value = userData[key] || '';
                }
            });
        }

        // 데이터 저장 처리
        function handleDataSave(e) {
            e.preventDefault();
            
            const formData = new FormData(e.target);
            const newData = {};
            
            // 모든 폼 필드 수집
            const inputs = e.target.querySelectorAll('input, select');
            inputs.forEach(input => {
                newData[input.id] = input.value;
            });
            
            // 변경 사항 추적
            const changes = trackDataChanges(userData, newData);
            
            // 데이터 업데이트
            userData = { ...newData };
            localStorage.setItem(`bni_data_${currentUser.email}`, JSON.stringify(userData));
            
            // 변경 내역 기록
            if (changes.length > 0) {
                addChangeHistory(changes);
                showChangePopup(changes);
            }
            
            // UI 업데이트
            updateSummary();
            showToast('데이터가 저장되었습니다!');
            
            // 구글 시트 동기화 시뮬레이션
            simulateGoogleSheetSync();
        }

        // 데이터 변경 사항 추적
        function trackDataChanges(oldData, newData) {
            const changes = [];
            
            Object.keys(newData).forEach(key => {
                if (oldData[key] !== newData[key]) {
                    changes.push({
                        field: key,
                        oldValue: oldData[key] || '',
                        newValue: newData[key] || '',
                        timestamp: new Date().toLocaleString('ko-KR')
                    });
                }
            });
            
            return changes;
        }

        // 변경 내역 추가
        function addChangeHistory(changes) {
            const historyEntry = {
                timestamp: new Date().toLocaleString('ko-KR'),
                changes: changes
            };
            
            changeHistory.unshift(historyEntry);
            
            // 최대 50개까지만 저장
            if (changeHistory.length > 50) {
                changeHistory = changeHistory.slice(0, 50);
            }
            
            localStorage.setItem(`bni_history_${currentUser.email}`, JSON.stringify(changeHistory));
        }

        // 변경 내역 팝업 표시
        function showChangePopup(changes) {
            const content = document.getElementById('changeContent');
            content.innerHTML = '';
            
            changes.forEach(change => {
                const changeDiv = document.createElement('div');
                changeDiv.className = 'p-3 bg-blue-50 rounded border-l-4 border-blue-400';
                changeDiv.innerHTML = `
                    <div class="text-sm font-medium text-gray-800">${getFieldLabel(change.field)}</div>
                    <div class="text-xs text-gray-600 mt-1">
                        <span class="line-through text-red-500">${change.oldValue || '(빈 값)'}</span>
                        <i class="fas fa-arrow-right mx-2 text-gray-400"></i>
                        <span class="text-green-600">${change.newValue || '(빈 값)'}</span>
                    </div>
                    <div class="text-xs text-gray-500 mt-1">${change.timestamp}</div>
                `;
                content.appendChild(changeDiv);
            });
            
            document.getElementById('changePopup').classList.remove('hidden');
        }

        // 팝업 닫기
        function closePopup() {
            document.getElementById('changePopup').classList.add('hidden');
        }

        // 요약 정보 업데이트
        function updateSummary() {
            let totalPartners = 0;
            let vStage = 0, cStage = 0, pStage = 0;
            
            for (let i = 1; i <= 4; i++) {
                const name = userData[`partner${i}Name`];
                const stage = userData[`partner${i}Stage`];
                
                if (name && name.trim()) {
                    totalPartners++;
                    if (stage === 'V') vStage++;
                    else if (stage === 'C') cStage++;
                    else if (stage === 'P') pStage++;
                }
            }
            
            // UI 업데이트
            document.getElementById('totalPartners').textContent = totalPartners;
            document.getElementById('vStagePartners').textContent = vStage;
            document.getElementById('cStagePartners').textContent = cStage;
            document.getElementById('pStagePartners').textContent = pStage;
            document.getElementById('currentPartners').textContent = pStage;
            document.getElementById('remainingPartners').textContent = Math.max(0, 4 - pStage);
            
            // 달성률 계산 및 표시
            const achievementRate = Math.min(100, (pStage / 4) * 100);
            document.getElementById('achievementPercent').textContent = Math.round(achievementRate) + '%';
            
            const ring = document.getElementById('achievementRing');
            ring.style.setProperty('--progress', achievementRate);
            
            // 마지막 업데이트 시간
            document.getElementById('lastUpdate').textContent = new Date().toLocaleString('ko-KR');
        }

            // 실제 구현에서는 Google Sheets API를 사용
            console.log('Google Sheets 동기화 시뮬레이션:', userData);
            showToast('구글 시트와 동기화되었습니다!');
        }

        // 동기화 버튼 처리
        function handleSync() {
            simulateGoogleSheetSync();
        }

        // 토스트 메시지 표시
        function showToast(message) {
            const toast = document.getElementById('successToast');
            const messageSpan = document.getElementById('toastMessage');
            
            messageSpan.textContent = message;
            toast.classList.remove('translate-x-full');
            
            setTimeout(() => {
                toast.classList.add('translate-x-full');
            }, 3000);
        }

        // 변경 사항 실시간 추적
        function trackChanges(e) {
            // 실시간으로 요약 정보 업데이트
            setTimeout(updateSummary, 100);
        }
